def create_model(input_dim = 3):
    model = Sequential([
               tf.keras.layers.Dense(units=16, 
                                     activation = 'relu', 
                                     kernel_initializer = 'normal',
                                     input_dim=input_dim),
               tf.keras.layers.Dropout(0.2),
               tf.keras.layers.Dense(units=16,
                                     activation = 'relu',
                                     kernel_initializer = 'normal'),
               tf.keras.layers.Dropout(0.2),
               tf.keras.layers.Dense(units=1, activation = 'sigmoid')])

    return model
